readonly

The readonly modifier ensures that the property won't be overwritten (except for the method's constructor). If a method tries to change it, an exception will be thrown.

class Program
{
	private readonly int _id;
	
	public Program(int id)
	{
		this._id = id;
	}

	public ChangeID(int id)
	{
		this._id = id; // Exception here
	}

	static void Main()
	{
		Program program = new Program(1);
		program.ChangeID(2);
	}
}
