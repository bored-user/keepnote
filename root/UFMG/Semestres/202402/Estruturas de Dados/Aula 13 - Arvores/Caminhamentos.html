Caminhamentos

Pre-ordem: visita (imprime) o item, depois visita recursivamente as subarvores a esquerda e a direita

In-ordem: visita recursivamente a esquerda, depois o item, depois recursivamente a direita

Pos-ordem: visita recursivamente a esquerda, depois recursivamente a direita, depois o no

Por nivel: visita os nodos de cada nivel em sequencia.

void PreOrdem(No* p) {
	if(p!=NULL) {
		p->item.imprime();
		PreOrdem(p->esq);
		PreOrdem(p->dir);
	}
}

void InOrdem(No* p) {
	if(p!=NULL) {
		InOrdem(p->esq);
		p->item.imprime();
		InOrdem(p->dir);
	}
}

void PosOrdem(No* p) {
	if(p!=NULL) {
		PosOrdem(p->esq);
		PosOrdem(p->dir);
		p->item.imprime();
	}
}

void CaminhaNivel() {
	Fila F;
	No* p;
	F.Enfileira(raiz);
	while(!F.Vazia()) {
		p = F.Desenfileira();
		if(p!=NULL) {
			p->item.Imprime();
			F.Enfileira(p->esq);
			F.Enfileira(p->dir);
		}
	}

}
