Take()

Takes an integer as argument. Returns a collection with the first n arguments of the receiver (callee).

using System;

class Program
{
	static void Main()
	{
		int[] nums = new int[] { 1, 2, 3 };
		Console.WriteLine(nums.Length); // 3
		Console.WriteLine(nums.Take(2).ToArray().Lenght); // 2
	}
}
