OrderBy()

Takes a lambda expression and orders the collection by the returned value of the given lambda expression:

using System.Collections.Generic;
using System.Linq;
using System;

class Book
{
	public string Title { get; set; }
	public double Price { get; set; }
}

class Program
{
	static void Main()
	{
		List<Book> books = new List<Book>() { new Book() { Title = "C# Guide", Price = 10.5 }, new Book() { Title = "Java Guide", Price = 10 }, new Book() { Title = "Python Guide", Price = 12 } };
		
		foreach(var book in books.OrderBy(b => b.Title).ToArray())
		{
			Console.WriteLine(book.Title);
			/*
				#=> C# Guide
				#=> Java Guide
				#=> Python Guide
			*/
		}
		
		foreach(var book in books.OrderBy(b => b.Price).ToArray())
		{
			Console.WriteLine(book.Price);
			/*
				#=> 10
				#=> 10.5
				#=> 12
			*/
		}
	}
}
