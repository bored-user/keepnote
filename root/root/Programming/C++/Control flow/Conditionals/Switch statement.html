Switch statement

#include <iostream>

int main() {
	int a = 101;

	switch(a) {
		case 10: {
				std::cout << "a is 10" << std::endl;
				break;
		}
	
		case 11: {
			std::cout << "a is 11" << std::endl;
			break;
		}
			
		default: {
			std::cout << "a != 10 & != 11" << std::endl;
			break;
		}
	}

	return 0;
}

 

Note that the switch expression must be of type int or enum.
