chomp

Remove new lines from the end of the string. Takes an optional parameter: specific character(s) to remove from the end of the string. If the string has no new lines & no parameter was given, the method returns the string unchanged.

str = 'a\n' # simulating user input
puts str.chomp #=> 'a'

str = 'a'
puts str.chomp #=> 'a'

str = 'joaoo'
puts str.chomp('o') #=> 'joao'

str = 'abc'
puts str.chomp('b') #=> 'abc', because it didn't find 'b' at the end of the string

str = 'abc'
puts str.chomp('bc') #=> 'a'
