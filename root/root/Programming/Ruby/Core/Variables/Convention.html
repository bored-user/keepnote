Convention

By convention, Ruby variable names follow the snake_case syntax.

They're also written in all lowercase, since uppercase variables are constants.

my_bool = true

name = 'Joao'
age = 123
male = true
money = nil
friends = [
    {
        'name' => 'Friend_1',
        'age' => 321
        'male' => true
        'money' => 232
    },
    {
        'name' => 'Friend_2',
        'age' => 213
        'male' => false
        'money' => 1
    },
    {
        'name' => 'Friend_3',
        'age' => 234
        'male' => false
        'money' => 999
    }
]
