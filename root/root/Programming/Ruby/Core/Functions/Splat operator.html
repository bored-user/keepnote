Splat operator

The splat operator is a way of getting every argument a function receives and join them into an Array.

It doesn't affect blocks.

 

def no_block_method(*args)
	p args
end

no_block_method #=> []
no_block_method(1, 2, 3) #=> [1, 2, 3]

def block_method(*args, &block)
	p args
	if block_given?
		block.call(args)
		yield args
	end
end

block_method #=> []
block_method(1, 2) #=> [1, 2]
block_method(1) {|args| p args} #=> [1] [1] [1]
