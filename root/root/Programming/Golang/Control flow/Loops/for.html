for

In Go, the for loop is the only existent loop. It syntax follows every other for loop syntax:

package main

import "fmt"

func main() {
	for x := 1; x < 3; x++ {
		fmt.print(x, " ")
		// 1 2
	}
	
	// It's also possible to loop using `var`
	for var x uint64 = 0; x < 999999999; x++ {
		fmt.printLn(x)
	}
	
	// Also, note that one can also provide only the condition:
	x := 0
	for x < 10 {
		fmt.printLn(x)
		// Without changing `x`, this would be an infinite loop
		x++
	}
}

Since there's no while loop, the only way to make an infinite loop is by omitting the for's operators:

package main

import "fmt"

func main() {
	for {
		fmt.printLn("infinite loop here!")	
	}
	
	// This also works:
	for true {
		fmt.printLn("infinite loop!")	
	}
}
