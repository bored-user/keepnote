switch

The switch statement, in Go, is a little different than on other languages.

package main

import "fmt"

func main() {
	name := "John"
	
	switch {
		case name == "John":
			fmt.Println("name == John")
	}
	
	switch name {
		case "John":
			fmt.Println("name == John")
	}
	
	switch name {
		case "John":
			fmt.Println("it's not necessary to use a `break` statement here!")
			
		case "Maria":
			fmt.Println("this line won't be executed!")
	}
	
	switch {
		case 1 == 2:
			fmt.Println("my computer doesn't know arithmetics")
			
		default:
			fmt.Println("it's also possible to use the default case!")
	}
	
	switch {
		case true:
			fmt.Print("one can use the `fallthrough` statement to ")
			fallthrough
			
		default:
			fmt.Println("make Go evaluate the next statement!")
	}
	
	switch name {
		case "John", "Maria":
			fmt.Println("`name` is either 'John' or 'Maria'")
	}
	
	var a interface{} = 5
	switch a.(type) {
		case string:
			fmt.Println("`a` is a string")
			
		case int:
			fmt.Println("`a` is an int")
			
		default:
			fmt.Println("`a` isn't a string nor an int")
	}
}

A few things to note:

  • switch {} is the same as switch true {}
  • Cases are indicated by case [condition]:
  • There's no need for a break statement, because Go, by default, executes only one case
  • One can use the default statement
  • One can use the fallthrough statement to make Go evaluate the next case
  • One can use a comma character to indicate an or relation
  • One can evaluate types using variable.(type) if variable is of type interface{}
