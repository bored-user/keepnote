Using types

In Go, variable types are indicated like so:

package main

import "fmt"

func main() {
	var name string = "John"
	var d int = 1
	var c int = 90
	
	fmt.printLn("I'm", name, "and I have", d, "dollar and", c, "cents in my wallet")
	// #=> "I'm John and I have 1 dollar and 90 cents in my wallet"
}
