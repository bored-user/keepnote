Min()

Returns the smallest number in a numeric collection. Takes an optional delegate indicating which property of the objects inside the collection should be numerically compared.

using System.Collections.Generic;
using System.Linq;
using System;

class Book
{
	public string Title { get; set; }
	public double Price { get; set; }
}

class Program
{
	static List<Book> GetBookList() => new List<Book>() { new Book() { Title = "C# Guide", Price = 10.5 }, new Book() { Title = "Java Guide", Price = 10 }, new Book() { Title = "Python Guide", Price = 12 } };

	static void Main()
	{
		Console.WriteLine(GetBookList().Min(b => b.Price))
		// Sorts by price. Prints "10"
	}
}
