Functions

The return type is specified  It can take any number of arguments (which are also type-defined by a space character). If a function returns void, one only need to set no return type (like the main function).

package main

import "fmt"

func sum(n1 int, n2 int) int {
	return n1 + n2
}

func fact(n int) int {
	if n == 1 {
		return 1
	} else {
		return fact(n - 1) * n
	}
}

func greet(name string) string {
	return fmt.Sprintf("Hello, %v", name)
}

func main() {
	fmt.Println(sum(10, 40))	// 50
	fmt.Println(fact(5))		// 120
	fmt.Println(greet("John"))	// "Hello, John"
}
