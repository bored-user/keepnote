Lists

Written between square brackets. Are able to store multiple data types, even other lists.
 

l = [1,"hi", 3.3]

print(l[0])			#=> 1
print(l[0:2])		#=> [1, "hi"]
print(l[-1])		#=> 3.3
print(l[1:])		#=> ["hi", 3.3]

print(len(l)) 		#=> 3

l.append("hello") 	# "hello" added as last item

l.pop() 			# removes the last item
l.pop(0) 			# removes the first item

ll = ["pair item1", "pair item2", " pair item3"]
lll = zip(l, ll)
print(list(lll)) 	#=> [(1, "pair item 1"), ("hi", "pair item 2"), (variable, "pair item 3")]
