Namespace resolution operator

The namespace resolution operator (::) is the way to access a Module's variables/ classes.

module Mod
    VERSION = '1.0.0'
    def self.say_hi
        'hi'
    end
    
    class ModuleClass
        def say_hello
            'hello'
        end
    end
end

class InheritClass < Mod::ModuleClass # inherits from ModuleClass from Mod
    def say_howdy
        'howdy'
    end
end

ic = InheritClass.new
mc = Mod::ModuleClass.new

puts Mod::VERSION #=> '1.0.0'
puts Mod.say_hi #=> 'hi'
puts ic.say_hello #=> 'hello'
puts ic.say_howdy #=> 'howdy'
puts mc.say_hello #=> 'hello'
