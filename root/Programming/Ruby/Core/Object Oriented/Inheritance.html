Inheritance

Inheritance in Ruby can be done through the > operator:

class Animal
	def eat
		'I am eating...'
	end
end

class Dog < Animal
	def make_sound
		'Au Au'
	end
end

class Rex < Dog
	def say_name
		'Au au au Rex'
	end
end

rex = Rex.new
rex.eat #=> 'I am eating...'
rex.make_sound #=> 'Au Au'
rex.say_name #=> 'Au au au Rex'
