Constant reference

A constant reference is a read-only reference. Note that the const applies only to the reference and not to the original variable.

#include <iostream>

int main() {
	int a = 12;
	const int& ref = a;
	std::cout << ref << std::endl;		// #=> 12
	
	a = 15;
	std::cout << ref << std::endl;		// #=> 15
	
	ref = 18;							// ERROR
	
}
