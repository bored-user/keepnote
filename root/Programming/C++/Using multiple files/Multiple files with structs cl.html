Multiple files with structs/ classes

// Student.hpp
#ifndef STUDENT_H
#define STUDENT_H
#include <iostream>
#include <string>

struct Student {
	int age;
	std::string name;
	
	Student(std::string name, int age);
	Student(std::string name);
	
	void say_hi(Student* s);
};
#endif
// Student.cpp
#include <iostream>
#include <string>

#include "Student.hpp"

Student::Student(std::string name, int age) {
	this->name = name;
	this->age = age;
}

Student::Student(std::string name) {
	this->name = name;
	this->age = 0;
}

void Student::say_hi(Student* s) {
	std::cout << "Hi, " << s->name << ", my name is " << this->name << std::endl;
}
// main.cpp
#include <iostream>
#include <string>

#include "Student.hpp"

int main() {
	Student s1 = new Student("Joao", 20);
	Student s2 = new Student("Maria");
	s1.say_hi(&s2);
	// Hi, Maria, my name is Joao
}

Notes:

  • only the .hpp file is included, both in the implementation file and usage file
  • in order to implement a method, one uses the :: operator
  • there's no {} at the end of function signatures in Student.hpp
  • STUDENT_H compilation flag - useful for avoiding multiple includes of the same file, overwriting compilation error
