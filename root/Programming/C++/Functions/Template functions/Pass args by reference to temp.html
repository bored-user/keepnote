Pass args by reference to template

One may template a function using reference arguments. It is generally a good practice to use constant references, so template <typename T>  const T& add(const T& a, const T& b) {  } (though it is not required, so template <typename T>  T& add(T& a, T& b) {  } would also work).

#include <iostream>

template <typename T>
const T& add(const T& a, const T& b) {
	return a + b;
}

template <typename T>
const &T max(const &T a, const &T b) {
	if(a > b) return a;
	else 	  return b;
}

int main() {
	int a1 = 10, b1 = 5;
	float a2 = 10.5, b2 = 5.5;
	double a3 = 10.5555555, b3 = 5.5555555;
	long a4 = 1000000000, b4 = 55555555555;
	
	int max1 = max(a1, b1);
	float max2 = max(a2, b2);
	double max3 = max(a3, b3);
	long max4 = max(a4, b4);
	
	std::cout << max1 << std::endl;
	std::cout << max2 << std::endl;
	std::cout << max3 << std::endl;
	std::cout << max4 << std::endl;

	return 0;
}
