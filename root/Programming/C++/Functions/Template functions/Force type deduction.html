Force type deduction

One may force the compiler to deduce one specific type using the <type> notation. On the add() below, one could call float add2 = add<float>(a2, b2); to force-deduce float.

#include <iostream>

template <typename T> T add(T a, T b) {
	return a + b;
}

int main() {
	int a1 = 10, b1 = 5;
	float a2 = 10.5, b2 = 5.5;
	double a3 = 10.5555555, b3 = 5.5555555;
	long a4 = 1000000000, b4 = 55555555555;
	
	int add1 = add(a1, b1);
	float add2 = add(a2, b2);
	double add3 = add(a3, b3);
	long add4 = add(a4, b4);
	
	std::cout << add1 << std::endl;
	std::cout << add2 << std::endl;
	std::cout << add3 << std::endl;
	std::cout << add4 << std::endl;
	
	return 0;
}
