Lambda functions

#include <iostream>

int main() {
	int sum = [](int a, int b) -> int { return a + b; };
	int result = sum(10, 20); // #=> 30
	
	int twoXthree = [](int a, int b) { return a * b }(10, 20);
	// return type is deduced by the compiler
	// #=> 200
	
	auto print = [](std::string str) { std::cout << str << std::endl; };
	print("hello, world!");	// #=> hello, world!
}
