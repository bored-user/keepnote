Select()

Takes a lambda expression. It selects only the property returned from the lambda expression, forming an array (of type <T>) out of it:

using System.Collections.Generic;
using System.Linq;
using System;

class Book
{
	public string Title { get; set; }
	public double Price { get; set; }
}

class Program
{
	static void Main()
	{
		List<Book> books = new List<Book>() { new Book() { Title = "C# Guide", Price = 10.5 }, new Book() { Title = "Java Guide", Price = 10 }, new Book() { Title = "Python Guide", Price = 12 } };
	}
}
