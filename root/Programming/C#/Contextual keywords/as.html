as

The as keyword tries to cast a type to another. If it fails, it returns null.

class ParentClass
{}

class ChildClass : ParentClass
{}

class Car
{}

class Program
{
	static void Main()
	{
		ChildClass cc = new ChildClass();
		
		ParentClass pc = cc; // This works
		Car c = pc; // this will fail because ParentClass has no relation to Car
		Car d = pc as Car; // `d` will be `null`
	}
}
